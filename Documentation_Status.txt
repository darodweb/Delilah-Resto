SELECT est.id AS ESTADO, p.hora AS HORA, p.id AS NUMERO, p.id AS DESCRIPCION,
fp.id AS METODO_DE_PAGO, p.total AS TOTAL, u.nombre AS USUARIO,
u.direccion AS DIRECCION FROM pedidos AS p
INNER JOIN usuarios AS u ON pedidos.usuario_id = u.id
INNER JOIN forma_de_pago as fp ON pedidos.forma_de_pago = fp.id
INNER JOIN estados as est ON pedidos.estado = est.id;
